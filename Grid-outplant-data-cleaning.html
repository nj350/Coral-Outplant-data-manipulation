Create columns which say if the monitoring event is 1 year, 2 year, 3 year or intermittent

#calculates duration between outplant and monitoring date
grids$Duration <- grids$Date - grids$Outplant.Date
#note some outplants have a duration of -2 at first outplant as the old database didn't differentiate between outplants over multiple days
#converts any durations below zero to zero

grids$Duration[grids$Duration < 0] <- 0

#creates a vector Period which corresponds to initial outplant, 1 year, 2 year, 3 year etc - uses the time period when the 1,2,3 year monitoring would take place
grids$Period <- ifelse(grids$Duration == 0, "0 Year",
                       ifelse(grids$Duration > 322 & grids$Duration < 405 |
                                grids$Duration == 425 & grids$Species == "AAGA"|
                                grids$Duration == 425 & grids$Species == "PAST"|
                                grids$Duration == 435 & grids$Species == "ACER", "1 Year", 
                         ifelse(grids$Duration > 684 & grids$Duration < 790, "2 Year",
                            ifelse(grids$Duration > 1020 & grids$Duration < 1200, "3 Year", ""))))
